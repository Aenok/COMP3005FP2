CREATE TABLE Member (
	m_id			SERIAL,
	f_name		VARCHAR(255) NOT NULL,
	l_name		VARCHAR(255) NOT NULL,
	email		VARCHAR(255) NOT NULL UNIQUE,
	password	VARCHAR(20) NOT NULL,
	height		INT,
	weight		INT,
	gender		TEXT,
	acc_balance	MONEY DEFAULT 0,
	PRIMARY KEY(m_id)
)


CREATE TABLE Mem_Goals (
	m_id		INT UNIQUE,
	t_weight	INT,
	t_cardio	INT,
	t_bench		INT,
	t_squat		INT,
	t_dl		INT,
	t_prot		INT,
	FOREIGN KEY(m_id)
		REFERENCES Member(m_id)
)

CREATE TABLE Achievements(
	m_id		INT UNIQUE,
	pr_weight	INT,
	pr_cardio	INT,
	pr_bench	INT,
	pr_dl		INT,
	pr_squat	INT,
	pr_prot		INT,
	FOREIGN KEY(m_id)
		REFERENCES Member(m_id)
)

CREATE TABLE Exercises(
	e_id		SERIAL,
	e_area		VARCHAR(20) NOT NULL,
	e_name		VARCHAR(50) UNIQUE,
	PRIMARY KEY(e_id)
)

CREATE TABLE mem_activity(
	m_id			INT UNIQUE,
	e_name			VARCHAR(50),
	dur				INT,
	reps			INT,
	weight_added	INT,
	e_date			DATE,
	FOREIGN KEY(m_id)
		REFERENCES Member(m_id),
	FOREIGN KEY(e_name)
		REFERENCES Exercises(e_name)
)
